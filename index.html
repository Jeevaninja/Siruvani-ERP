-- Run in: Supabase Dashboard → SQL Editor

-- 1. Add missing columns to sales_entries
ALTER TABLE sales_entries
  ADD COLUMN IF NOT EXISTS express_sales numeric DEFAULT 0,
  ADD COLUMN IF NOT EXISTS cooldrink_sales numeric DEFAULT 0,
  ADD COLUMN IF NOT EXISTS cigarette_sales numeric DEFAULT 0,
  ADD COLUMN IF NOT EXISTS sales_file_url text;

-- 2. Fix purchases table (type → purchase_type, avoids reserved word)
ALTER TABLE purchases
  ADD COLUMN IF NOT EXISTS purchase_type text DEFAULT 'liquor';

-- 3. Create expense_categories table (was missing entirely)
CREATE TABLE IF NOT EXISTS expense_categories (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users(id) ON DELETE CASCADE,
  categories jsonb DEFAULT '{}',
  created_at timestamptz DEFAULT now(),
  UNIQUE(user_id)
);
ALTER TABLE expense_categories ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "user_own_cats" ON expense_categories;
CREATE POLICY "user_own_cats" ON expense_categories
  FOR ALL USING (auth.uid() = user_id);

-- 4. Ensure expenses has sub_category column
ALTER TABLE expenses
  ADD COLUMN IF NOT EXISTS sub_category text;

-- 5. Ensure stock_entries exists
CREATE TABLE IF NOT EXISTS stock_entries (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users(id) ON DELETE CASCADE,
  entry_date date NOT NULL,
  opening_stock numeric DEFAULT 0,
  closing_stock numeric DEFAULT 0,
  created_at timestamptz DEFAULT now(),
  UNIQUE(user_id, entry_date)
);
ALTER TABLE stock_entries ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "user_own_stock" ON stock_entries;
CREATE POLICY "user_own_stock" ON stock_entries
  FOR ALL USING (auth.uid() = user_id);

-- 6. Fix purchase_type check constraint (blocks inserts)
DO $$
BEGIN
  -- Drop old check constraint if it exists
  IF EXISTS (
    SELECT 1 FROM information_schema.constraint_column_usage
    WHERE table_name = 'purchases' AND constraint_name = 'purchases_purchase_type_check'
  ) THEN
    ALTER TABLE purchases DROP CONSTRAINT purchases_purchase_type_check;
  END IF;
  -- Drop any other check constraints on purchase_type
  EXECUTE (
    SELECT string_agg('ALTER TABLE purchases DROP CONSTRAINT ' || quote_ident(constraint_name) || ';', ' ')
    FROM information_schema.table_constraints
    WHERE table_name = 'purchases'
      AND constraint_type = 'CHECK'
      AND constraint_name ILIKE '%type%'
  );
EXCEPTION WHEN OTHERS THEN NULL;
END $$;

-- Ensure purchase_type column has no constraint
ALTER TABLE purchases ALTER COLUMN purchase_type DROP DEFAULT;
ALTER TABLE purchases ALTER COLUMN purchase_type SET DEFAULT 'liquor';

-- Done! Refresh the app after running.
